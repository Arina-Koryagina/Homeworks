Змінна -- іменована частина пам'яті, яка зберігає значення
Логіка -- наука про твердження

a..z, A..Z, 0..9, _ тільки!
_1a (без пробілів, цифри не можуть бути попереду)

variable (змінна/має місце) = literal (літерал/не має місця)
 = -- присвоєння
c = -a -- унарний мінус
c = +a -- унарний плюс
not -- унарний оператор логічної інверсїі, зазначається перед змінною
 > < == != <= >= or (||) and (&&) -- бінарні оператори логіки
 - + унарні
 = + - * / ** % // += -= *= /= **= %= //= бінарні
 % -- остача від ділення
 // -- ціла частина від ділення
 a = a + b # a += b операція з одночасним присвоєнням
abs() -- модуль
sqrt() -- корінь квадратний
a, b = b, a -- кортеж

int = integer (ціле число)
float = fraction number (дробові числа)
str = string (рядок/текст)
bool = boolean = True (1) / False (0), (бінарна логіка)

\t - таб
\n - перенос тексту
